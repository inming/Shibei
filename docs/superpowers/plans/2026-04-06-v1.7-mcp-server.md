```rust
/// Get the plain text content for a resource. Returns None if not yet extracted.
pub fn get_plain_text(conn: &Connection, resource_id: &str) -> Result<Option<String>, DbError> {
    let mut stmt = conn.prepare(
        "SELECT plain_text FROM resources WHERE id = ?1 AND deleted_at IS NULL",
    )?;
    let text = stmt.query_row(params![resource_id], |row| row.get::<_, Option<String>>(0))
        .map_err(|e| match e {
            rusqlite::Error::QueryReturnedNoRows => {
                DbError::NotFound(format!("resource not found: {}", resource_id))
            }
            other => DbError::Sqlite(other),
        })?;
    Ok(text)
}

/// Set the plain text content for a resource.
pub fn set_plain_text(conn: &Connection, resource_id: &str, text: &str) -> Result<(), DbError> {
    let rows = conn.execute(
        "UPDATE resources SET plain_text = ?1 WHERE id = ?2 AND deleted_at IS NULL",
        params![text, resource_id],
    )?;
    if rows == 0 {
        return Err(DbError::NotFound(format!("resource not found: {}", resource_id)));
    }
    Ok(())
}
```

```rust
use scraper::{Html, Selector};

/// Tags whose text content should be excluded from plain text output.
const EXCLUDED_TAGS: &[&str] = &["script", "style", "noscript", "svg", "math"];

/// Block-level tags that should produce paragraph breaks.
const BLOCK_TAGS: &[&str] = &[
    "p", "div", "section", "article", "aside", "header", "footer", "nav", "main",
    "h1", "h2", "h3", "h4", "h5", "h6", "blockquote", "pre", "ul", "ol", "li",
    "table", "tr", "td", "th", "figure", "figcaption", "details", "summary",
    "br", "hr",
];

pub fn extract_plain_text(html: &str) -> String {
    if html.is_empty() {
        return String::new();
    }

    let document = Html::parse_document(html);

    // Build selectors for excluded tags
    let excluded_selector = Selector::parse(&EXCLUDED_TAGS.join(",")).unwrap();
    let excluded_ids: std::collections::HashSet<scraper::ElementRef> = std::collections::HashSet::new();

    // Collect IDs of excluded elements and their descendants
    let mut excluded_node_ids: std::collections::HashSet<ego_tree::NodeId> =
        std::collections::HashSet::new();
    for el in document.select(&excluded_selector) {
        // Mark this element and all descendants as excluded
        for descendant in el.id().descendants(&document.tree) {
            excluded_node_ids.insert(descendant);
        }
    }
    drop(excluded_ids);

    let block_selector = Selector::parse(&BLOCK_TAGS.join(",")).unwrap();
    let block_node_ids: std::collections::HashSet<ego_tree::NodeId> = document
        .select(&block_selector)
        .map(|el| el.id())
        .collect();

    let mut result = String::new();
    let mut last_was_block = false;

    for edge in document.tree.root().traverse() {
        match edge {
            ego_tree::iter::Edge::Open(node) => {
                if let Some(el) = node.value().as_element() {
                    if block_node_ids.contains(&node.id()) && !result.is_empty() && !last_was_block {
                        result.push('\n');
                        last_was_block = true;
                    }
                    // Skip hidden elements
                    if el.attr("hidden").is_some() {
                        continue;
                    }
                }
                if let Some(text) = node.value().as_text() {
                    if excluded_node_ids.contains(&node.id()) {
                        continue;
                    }
                    // Skip if any ancestor is excluded
                    let mut ancestor = node.parent();
                    let mut skip = false;
                    while let Some(a) = ancestor {
                        if excluded_node_ids.contains(&a.id()) {
                            skip = true;
                            break;
                        }
                        ancestor = a.parent();
                    }
                    if skip {
                        continue;
                    }

                    let t = text.trim();
                    if !t.is_empty() {
                        if !result.is_empty() && !last_was_block {
                            result.push(' ');
                        }
                        result.push_str(t);
                        last_was_block = false;
                    }
                }
            }
            ego_tree::iter::Edge::Close(node) => {
                if block_node_ids.contains(&node.id()) && !result.is_empty() {
                    if !result.ends_with('\n') {
                        result.push('\n');
                    }
                    last_was_block = true;
                }
            }
        }
    }

    // Clean up: collapse multiple newlines into double, trim
    let mut cleaned = String::with_capacity(result.len());
    let mut newline_count = 0;
    for ch in result.chars() {
        if ch == '\n' {
            newline_count += 1;
            if newline_count <= 2 {
                cleaned.push('\n');
            }
        } else {
            newline_count = 0;
            cleaned.push(ch);
        }
    }

    cleaned.trim().to_string()
}
```

```rust
async fn handle_list_resources(
    State(state): State<Arc<AppState>>,
    headers: HeaderMap,
    axum::extract::Query(query): axum::extract::Query<ResourcesQuery>,
) -> Result<Json<Vec<resources::Resource>>, (StatusCode, Json<ErrorResponse>)> {
    verify_token(&headers, &state.token)?;

    let conn = state.pool.get().map_err(|e| {
        (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() }))
    })?;

    let tag_ids: Vec<String> = query
        .tag_ids
        .as_deref()
        .unwrap_or("")
        .split(',')
        .filter(|s| !s.is_empty())
        .map(|s| s.to_string())
        .collect();

    let sort_by = match query.sort_by.as_deref() {
        Some("annotated_at") => resources::SortBy::AnnotatedAt,
        _ => resources::SortBy::CreatedAt,
    };
    let sort_order = match query.sort_order.as_deref() {
        Some("asc") => resources::SortOrder::Asc,
        _ => resources::SortOrder::Desc,
    };

    let results = if let Some(q) = &query.query {
        if q.len() >= 2 {
            search::search_resources(
                &conn,
                q,
                query.folder_id.as_deref(),
                &tag_ids,
                query.sort_by.as_deref().unwrap_or("created_at"),
                query.sort_order.as_deref().unwrap_or("desc"),
            )
        } else {
            Ok(Vec::new())
        }
    } else if let Some(folder_id) = &query.folder_id {
        resources::list_resources_by_folder(&conn, folder_id, sort_by, sort_order, &tag_ids)
    } else {
        resources::list_all_resources(&conn, sort_by, sort_order, &tag_ids)
    };

    results.map(Json).map_err(|e| {
        (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() }))
    })
}
```

```rust
    headers: HeaderMap,
    axum::extract::Path(id): axum::extract::Path<String>,
```

```rust
async fn handle_get_content(
    State(state): State<Arc<AppState>>,
    headers: HeaderMap,
    axum::extract::Path(id): axum::extract::Path<String>,
    axum::extract::Query(query): axum::extract::Query<ContentQuery>,
) -> Result<Json<ContentResponse>, (StatusCode, Json<ErrorResponse>)> {
    verify_token(&headers, &state.token)?;

    let conn = state.pool.get().map_err(|e| {
        (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() }))
    })?;

    // Verify resource exists
    resources::get_resource(&conn, &id).map_err(|e| match &e {
        db::DbError::NotFound(_) => (StatusCode::NOT_FOUND, Json(ErrorResponse { error: e.to_string() })),
        _ => (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() })),
    })?;

    // Try to get plain_text, lazy-fill if missing
    let mut text = resources::get_plain_text(&conn, &id).map_err(|e| {
        (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() }))
    })?;

    if text.is_none() {
        // Try to extract from snapshot file
        let snapshot_path = state.base_dir.join("storage").join(&id).join("snapshot.html");
        if snapshot_path.exists() {
            if let Ok(html) = std::fs::read_to_string(&snapshot_path) {
                let extracted = plain_text::extract_plain_text(&html);
                if !extracted.is_empty() {
                    let _ = resources::set_plain_text(&conn, &id, &extracted);
                    text = Some(extracted);
                }
            }
        }
    }

    let full_text = match text {
        Some(t) => t,
        None => {
            return Err((
                StatusCode::NOT_FOUND,
                Json(ErrorResponse {
                    error: "Content not available. Please open this resource in Shibei app to download the snapshot first.".to_string(),
                }),
            ));
        }
    };

    let offset = query.offset.unwrap_or(0);
    let max_length = query.max_length.unwrap_or(50000);
    let total_length = full_text.chars().count();

    let content: String = full_text.chars().skip(offset).take(max_length).collect();
    let has_more = offset + max_length < total_length;

    Ok(Json(ContentResponse {
        content,
        total_length,
        has_more,
    }))
}
```

```rust
async fn handle_update_resource(
    State(state): State<Arc<AppState>>,
    headers: HeaderMap,
    axum::extract::Path(id): axum::extract::Path<String>,
    Json(payload): Json<UpdateResourceRequest>,
) -> Result<Json<serde_json::Value>, (StatusCode, Json<ErrorResponse>)> {
    verify_token(&headers, &state.token)?;

    let conn = state.pool.get().map_err(|e| {
        (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() }))
    })?;

    let sync_ctx = state.sync_context();

    // Get current resource for defaults
    let current = resources::get_resource(&conn, &id).map_err(|e| match &e {
        db::DbError::NotFound(_) => (StatusCode::NOT_FOUND, Json(ErrorResponse { error: e.to_string() })),
        _ => (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() })),
    })?;

    // Update metadata if title or description provided
    if payload.title.is_some() || payload.description.is_some() {
        let title = payload.title.as_deref().unwrap_or(&current.title);
        let description = if payload.description.is_some() {
            payload.description.as_deref()
        } else {
            current.description.as_deref()
        };
        resources::update_resource(&conn, &id, title, description, sync_ctx.as_ref()).map_err(|e| {
            (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() }))
        })?;
    }

    // Move to different folder if folder_id provided
    if let Some(ref folder_id) = payload.folder_id {
        resources::move_resource(&conn, &id, folder_id, sync_ctx.as_ref()).map_err(|e| {
            (StatusCode::INTERNAL_SERVER_ERROR, Json(ErrorResponse { error: e.to_string() }))
        })?;
    }

    let _ = state.app_handle.emit(events::DATA_RESOURCE_CHANGED, serde_json::json!({
        "action": "updated",
        "resource_id": id,
        "folder_id": payload.folder_id.as_deref().unwrap_or(&current.folder_id),
    }));

    Ok(Json(serde_json::json!({ "ok": true })))
}
```

```typescript
import { readFileSync } from "node:fs";
import { join } from "node:path";
import { homedir } from "node:os";

const BASE_URL = "http://127.0.0.1:21519";

function getTokenPath(): string {
  if (process.env.SHIBEI_DATA_DIR) {
    return join(process.env.SHIBEI_DATA_DIR, "mcp-token");
  }
  // macOS default
  return join(homedir(), "Library", "Application Support", "shibei", "mcp-token");
}

function readToken(): string {
  try {
    return readFileSync(getTokenPath(), "utf-8").trim();
  } catch {
    throw new Error(
      "Shibei app is not running. Please start the app first."
    );
  }
}

export class ShibeiClient {
  private token: string;

  constructor() {
    this.token = readToken();
  }

  async request<T>(
    method: string,
    path: string,
    body?: unknown
  ): Promise<T> {
    const url = `${BASE_URL}${path}`;

    let response: Response;
    try {
      response = await fetch(url, {
        method,
        headers: {
          Authorization: `Bearer ${this.token}`,
          "Content-Type": "application/json",
        },
        body: body ? JSON.stringify(body) : undefined,
      });
    } catch {
      throw new Error(
        "Shibei app is not running. Please start the app first."
      );
    }

    if (response.status === 401) {
      throw new Error(
        "Authentication failed. Token may be stale, please restart Shibei app."
      );
    }

    if (!response.ok) {
      const err = (await response.json().catch(() => ({
        error: `HTTP ${response.status}`,
      }))) as { error: string };
      throw new Error(err.error);
    }

    return (await response.json()) as T;
  }

  async get<T>(path: string): Promise<T> {
    return this.request<T>("GET", path);
  }

  async post<T>(path: string, body?: unknown): Promise<T> {
    return this.request<T>("POST", path, body);
  }

  async put<T>(path: string, body?: unknown): Promise<T> {
    return this.request<T>("PUT", path, body);
  }

  async delete<T>(path: string): Promise<T> {
    return this.request<T>("DELETE", path);
  }
}
```

```typescript
    "search_resources",

    {
      query: z.string().optional().describe("Search keyword (>= 2 characters)"),
      folder_id: z.string().optional().describe("Filter by folder ID"),

        .optional()
        .describe("Filter by tag IDs (OR logic — matches resources with any of these tags)"),

        .enum(["created_at", "annotated_at"])
        .optional()

        .enum(["asc", "desc"])
        .optional()

      if (params.tag_ids?.length) queryParts.push(`tag_ids=${params.tag_ids.join(",")}`);

      if (params.query) queryParts.push(`query=${encodeURIComponent(params.query)}`);

            text: resources.length > 0
```

```typescript
import type { AnnotationsResponse } from "../types.js";

export function registerAnnotationTools(server: McpServer, client: ShibeiClient) {
  server.tool(
    "get_annotations",
    "Get all highlights and comments/notes for a resource.",
    {
      resource_id: z.string().describe("The resource ID"),
    },
    async (params) => {
      const data = await client.get<AnnotationsResponse>(
        `/api/resources/${encodeURIComponent(params.resource_id)}/annotations`

      const parts: string[] = [];

      if (data.highlights.length > 0) {
        parts.push("## Highlights\n");
        for (const h of data.highlights) {
          parts.push(`- [${h.color}] "${h.text_content}" (id: ${h.id})`);
          // Find comments for this highlight
          const relatedComments = data.comments.filter((c) => c.highlight_id === h.id);
          for (const c of relatedComments) {
            parts.push(`  Comment (id: ${c.id}): ${c.content}`);
          }
        }
      }

      // Resource-level notes (comments without highlight_id)
      const notes = data.comments.filter((c) => c.highlight_id === null);
      if (notes.length > 0) {
        parts.push("\n## Notes\n");
        for (const n of notes) {
          parts.push(`- (id: ${n.id}, ${n.updated_at}): ${n.content}`);
        }
      }

      const text =
        parts.length > 0
          ? parts.join("\n")
          : "No annotations found for this resource.";

      return { content: [{ type: "text" as const, text }] };
    }
  );
```

```typescript
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { ShibeiClient } from "../client.js";
import type { FolderNode } from "../types.js";

export function registerFolderTools(server: McpServer, client: ShibeiClient) {
  server.tool(
    "list_folders",
    "Get the folder tree structure of the resource library, including resource counts per folder.",
    {},
    async () => {
      const [folders, counts] = await Promise.all([
        client.get<FolderNode[]>("/api/folders"),
        client.get<Record<string, number>>("/api/folder-counts"),
      ]);

      function formatTree(nodes: FolderNode[], indent: number): string {
        return nodes
          .map((n) => {
            const prefix = "  ".repeat(indent);
            const count = counts[n.id] ?? 0;
            const line = `${prefix}- ${n.name} (id: ${n.id}, ${count} resources)`;
            const children =
              n.children.length > 0 ? "\n" + formatTree(n.children, indent + 1) : "";
            return line + children;
          })
          .join("\n");
      }

      const text =
        folders.length > 0
          ? `Folder tree:\n${formatTree(folders, 0)}`
          : "No folders found.";

      return { content: [{ type: "text" as const, text }] };
    }
  );
}
```
